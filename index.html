How it works: animation and rendering notes

The animated grid above is drawn with Canvas 2D to emulate a lightweight WebGL feel without external deps. We render a perspective grid with sinusoidal Z-warping and a camera pan to suggest “context expansion.”

// Canvas animation (WebGL-style grid)
const cvs = document.getElementById('grid');
const ctx = cvs.getContext('2d');
let t = 0; const DPR = Math.min(2, window.devicePixelRatio||1);
function resize(){ cvs.width = cvs.clientWidth * DPR; cvs.height = cvs.clientHeight * DPR; ctx.setTransform(DPR,0,0,DPR,0,0); }
window.addEventListener('resize', resize); resize();
function proj(x,y,z){ const f= 260/(z+320); return [x*f + cvs.width/(2*DPR), y*f + cvs.height/(1.4*DPR)]; }
function frame(){ t+=0.016; ctx.clearRect(0,0,cvs.width,DPR?cvs.height:0);
  ctx.strokeStyle = 'rgba(0,209,255,0.35)'; ctx.lineWidth=1;
  for(let i=-10;i<=10;i++){ ctx.beginPath(); for(let z=0; z<=600; z+=12){ const zz = z + Math.sin((z*0.02)+t)*14; const [px,py]=proj(i*22, 0, zz); if(z==0) ctx.moveTo(px,py); else ctx.lineTo(px,py);} ctx.stroke(); }
  for(let j=0;j<=22;j++){ ctx.beginPath(); for(let z=0; z<=600; z+=12){ const zz = z + Math.sin((z*0.02)+t+1.2)*14; const [px,py]=proj(-220 + j*20, 0, zz); if(z==0) ctx.moveTo(px,py); else ctx.lineTo(px,py);} ctx.stroke(); }
  const sweepY = Math.sin(t*0.6)*40; ctx.strokeStyle='rgba(255,77,79,0.45)';
  ctx.beginPath(); for(let x=-220; x<=220; x+=8){ const [px,py]=proj(x, sweepY, 240); if(x==-220) ctx.moveTo(px,py); else ctx.lineTo(px,py);} ctx.stroke();
  requestAnimationFrame(frame); } requestAnimationFrame(frame);

Why Canvas 2D? For a release landing, clarity + perf beat heavy shader stacks. The math mimics a vertex shader: perspective projection + time‑varying displacement. Swap to WebGL easily by porting proj to a mat4 MVP and drawing indexed strips.
